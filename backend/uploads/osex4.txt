Round Robin:

#include <stdio.h>

struct Process {
    int pid;
    int burst;
    int priority;
    int waiting;
    int turnaround;
    int rem_burst; // For RR
};

void displayGanttChart(int gantt[], int count, int completion_times[]) {
    printf("\nGantt Chart:\n ");
    for(int i = 0; i < count; i++) printf("--");
    printf("\n|");
    for(int i = 0; i < count; i++) printf("P%d|", gantt[i]);
    printf("\n ");
    for(int i = 0; i < count; i++) printf("--");
    printf("\n0");
    for(int i = 0; i < count; i++) printf("%d", completion_times[i]);
    printf("\n");
}

int main() {
    int n, tq;
    printf("Enter the number of processes: ");
    scanf("%d", &n);
    struct Process processes[n];

    int completed = 0;
    for(int i = 0; i < n; i++) {
        printf("\nEnter details for process %d:\n", i + 1);
        processes[i].pid = i + 1;
        printf("CPU Burst Time: ");
        scanf("%d", &processes[i].burst);
        printf("Priority: ");
        scanf("%d", &processes[i].priority);
        processes[i].rem_burst = processes[i].burst;
        processes[i].waiting = 0;
    }
    printf("Enter Time Quantum: ");
    scanf("%d", &tq);

    int t = 0, flag, i = 0, gc[100], gcc = 0, ct;
    while(completed < n) {
        flag = 0;
        if(processes[i].rem_burst > 0) {
            flag = 1;
            gc[gcc] = processes[i].pid;
            if(processes[i].rem_burst > tq) {
                t += tq;
                processes[i].rem_burst -= tq;
            } else {
                t += processes[i].rem_burst;
                processes[i].waiting = t - processes[i].burst;
                processes[i].turnaround = t;
                processes[i].rem_burst = 0;
                completed++;
            }
            ct[gcc] = t;
            gcc++;
        }
        i = (i+1)%n;
        if(!flag && completed < n) {
            i = (i+1)%n;
        }
    }
    printf("\nProcess\tBurst\tPriority\tWaiting\tTurnaround\n");
    for(int i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t\t%d\t%d\n",
               processes[i].pid, processes[i].burst, processes[i].priority,
               processes[i].waiting, processes[i].turnaround);
    }
    displayGanttChart(gc, gcc, ct);
    return 0;
}


Priority:

#include <stdio.h>
#include <stdlib.h>

struct Process {
    int pid;
    int burst;
    int priority;
    int waiting;
    int turnaround;
};

void displayGanttChart(struct Process processes[], int n) {
    printf("\nGantt Chart:\n ");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < processes[i].burst; j++) printf("--");
        printf(" ");
    }
    printf("\n|");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < processes[i].burst - 1; j++) printf(" ");
        printf("P%d", processes[i].pid);
        for(int j = 0; j < processes[i].burst - 1; j++) printf(" ");
        printf("|");
    }
    printf("\n ");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < processes[i].burst; j++) printf("--");
        printf(" ");
    }
    printf("\n0");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < processes[i].burst; j++) printf("  ");
        printf("%d", processes[i].turnaround);
    }
    printf("\n");
}

// Compare function for Priority
int cmpPriority(const void* a, const void* b) {
    // Lower value means higher priority
    return ((struct Process*)a)->priority - ((struct Process*)b)->priority;
}

int main() {
    int n;
    printf("Enter the number of processes: ");
    scanf("%d", &n);
    struct Process processes[n];

    // Input
    for(int i = 0; i < n; i++) {
        printf("\nEnter details for process %d:\n", i + 1);
        processes[i].pid = i + 1;
        printf("CPU Burst Time: ");
        scanf("%d", &processes[i].burst);
        printf("Priority: ");
        scanf("%d", &processes[i].priority);
    }

    // Priority Scheduling: Sort by priority
    qsort(processes, n, sizeof(struct Process), cmpPriority);

    // Calculate waiting & turnaround
    int current_time = 0;
    for(int i = 0; i < n; i++) {
        processes[i].waiting = current_time;
        processes[i].turnaround = current_time + processes[i].burst;
        current_time += processes[i].burst;
    }

    // Display results
    printf("\nProcess\tBurst\tPriority\tWaiting\tTurnaround\n");
    for(int i = 0; i < n; i++) {
        printf("P%d\t%d\t%d\t\t%d\t%d\n",
               processes[i].pid, processes[i].burst, processes[i].priority,
               processes[i].waiting, processes[i].turnaround);
    }
    displayGanttChart(processes, n);
    return 0;
}
